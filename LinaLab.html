// Генериране на пореден номер

function generateOrderNumber() {

    const now = new Date();

    const year = now.getFullYear();

    const month = (now.getMonth() + 1).toString().padStart(2, '0');

    

const currentMonth = `${year}${month}`;

const storedMonth = localStorage.getItem('currentMonth');


if (storedMonth !== currentMonth) {

localStorage.setItem('currentMonth', currentMonth);

localStorage.setItem('lastNumber', '0');

}


let number = parseInt(localStorage.getItem('lastNumber') || '0') + 1;

const formattedNumber = number.toString().padStart(3, '0');


document.getElementById('orderNumber').textContent = `${currentMonth}-${formattedNumber}`;

localStorage.setItem('lastNumber', number.toString());

}


// Настройка на маска за дата

function setupDateMask() {

const birthdateInput = document.getElementById('birthdate');

birthdateInput.addEventListener('input', function(e) {

let value = e.target.value.replace(/\D/g, '');

if (value.length >= 2) value = value.slice(0,2) + '.' + value.slice(2);

if (value.length >= 5) value = value.slice(0,5) + '.' + value.slice(5);

if (value.length > 10) value = value.slice(0,10);

e.target.value = value;


if (value.length === 10) calculateAge();

});

}


// Изчисляване на възраст

function calculateAge() {

const birthdateInput = document.getElementById('birthdate');

const birthdate = birthdateInput.value;


if (!birthdate || !/^\d{2}\.\d{2}\.\d{4}$/.test(birthdate)) {

document.getElementById('age').textContent = '0 години';

return;

}


const [day, month, year] = birthdate.split('.');

const birthDate = new Date(year, month - 1, day);

const today = new Date();


if (birthDate > today) {

alert('Датата на раждане не може да бъде в бъдещето');

birthdateInput.value = '';

return;

}


let age = today.getFullYear() - birthDate.getFullYear();

const monthDiff = today.getMonth() - birthDate.getMonth();


if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birthDate.getDate())) {

age--;

}


document.getElementById('age').textContent = `${age} години`;

}

// Добавяне на тест

function addTest() {

const testsSelect = document.getElementById('tests');

if (!testsSelect.value) return;


const test = JSON.parse(testsSelect.value);


if (selectedTests.some(t => t.name === test.name)) {

showMessage('Това изследване вече е добавено!', 'error');

return;

}


selectedTests.push(test);

updateTestList();

calculateTotal();

}


// Обновяване на списъка с тестове

<button type="button" class="remove-test no-print" onclick="removeTest(${index})">×</button>

`;

});

}


// Премахване на тест

function removeTest(index) {

selectedTests.splice(index, 1);

updateTestList();

calculateTotal();

}

}


// Изчисляване на общата сума

function calculateTotal() {

totalCost = selectedTests.reduce((sum, test) => sum + test.price, 0);

document.getElementById('totalCost').textContent = totalCost.toFixed(2);


let finalCost = totalCost;

const discountCheck = document.getElementById('discountCheck');

if (discountCheck.checked) {

const discount = parseInt(document.getElementById('discountAmount').value);

finalCost = totalCost * (1 - discount / 100);

}


document.getElementById('finalCost').textContent = finalCost.toFixed(2);

}

discountAmount.style.display = document.getElementById('discountCheck').checked ? 'inline' : 'none';

calculateTotal();

}
